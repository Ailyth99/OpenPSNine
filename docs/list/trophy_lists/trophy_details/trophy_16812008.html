  • kouhu_2
    看到别人用连发手柄自动刷钱真开心，没有手柄只好研究了下ahk，写了个简单的自动刷钱脚本，有兴趣的可以试试，remoteplay可以放在后台运行，别忘了游戏里把R1那个键改为圆圈， 用来放加速器碎片
    下载个autohotkey装好，把下面的脚本保存成haha.ahk文件然后运行，按ctrl+j启动，ESC停止
    适配的是下水道存档点正上方三个房间那条路，人物站在最左边的那个房间启动即可，如果喜欢其他地方修改sleep时间就行了
    原理就是模拟按住加速器来回跑，第一次写ahk，语法看着比较挫
    SetKeyDelay, 10, 10
    ^j::
    Loop
    {
    ControlSend,,{Enter down}, PS4遥控操作
    ControlSend,,{Right down}, PS4遥控操作
    sleep 1500
    ControlSend,,{Right up}, PS4遥控操作
    ControlSend,,{Left down}, PS4遥控操作
    sleep 1600
    ControlSend,,{Left up}, PS4遥控操作
    ControlSend,,{Enter up}, PS4遥控操作
    sleep 500
    }
    return

    esc::
    ControlSend,,{Right up}, PS4遥控操作
    ControlSend,,{Left up}, PS4遥控操作
    ControlSend,,{Enter up}, PS4遥控操作
    pause
    return

    发个crounmax用的脚本,装备加速,等级升到9x9,侦探眼睛升满可以自动拾取道具
    在如图位置,左边有石头挡住,按L3开始,R3停止


    int onoff;


    main {
    if(event_press(PS4_L3)) onoff=!onoff;
    if(onoff){combo_run(run);}
    if(event_press(PS4_R3)) {onoff=!onoff;combo_stop(run)}
    }

    combo run {
    set_val(PS4_R1,100);
    set_val(PS4_RIGHT,100);
    wait(1180);
    set_val(PS4_R1,0);
    set_val(PS4_RIGHT,0);

    wait(100);

    set_val(PS4_R1,100);
    set_val(PS4_LEFT,100);
    wait(1350);
    set_val(PS4_R1,0);
    set_val(PS4_LEFT,0);

    wait(100);
    }
